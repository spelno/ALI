(Gunakan variabel SS dan SHEETS dari kode sebelumnya) ... // Tambahkan baris ini di bagian konstanta SHEETS: // PENGELUARAN: SS.getSheetByName("Pengeluaran") function doPost(e) { let request = JSON.parse(e.postData.contents); const action = request.action; const ss = SpreadsheetApp.getActiveSpreadsheet(); // --- ACTION: GET ADMIN DASHBOARD (Data Lengkap) --- if (action === "getAdminData") { const prd = ss.getSheetByName("Produk").getDataRange().getValues(); const trx = ss.getSheetByName("Transaksi").getDataRange().getValues(); const exp = ss.getSheetByName("Pengeluaran").getDataRange().getValues(); // Ambil 20 transaksi terakhir let recentTrx = trx.slice(-20).reverse().map(r => ({ tgl: Utilities.formatDate(new Date(r[0]), "GMT+8", "dd/MM HH:mm"), nama: r[4], total: r[7] })); // Ambil 20 pengeluaran terakhir let recentExp = exp.slice(-20).reverse().map(r => ({ tgl: Utilities.formatDate(new Date(r[0]), "GMT+8", "dd/MM"), ket: r[1], jumlah: r[2] })); return createResponse({ status: "success", products: prd.slice(1), // Semua produk history: recentTrx, expenses: recentExp }); } // --- ACTION: ADD EXPENSE (Catat Pengeluaran) --- if (action === "addExpense") { const sheet = ss.getSheetByName("Pengeluaran"); sheet.appendRow([new Date(), request.keterangan, request.jumlah, request.admin]); return createResponse({ status: "success", message: "Pengeluaran dicatat!" }); } // ... (Sertakan juga action login dan getProducts dari script sebelumnya) ... } function createResponse(data) { return ContentService.createTextOutput(JSON.stringify(data)).setMimeType(ContentService.MimeType.JSON); }
