/** * KONFIGURASI GLOBAL */ const SHEET_PRODUK = "Produk"; const SHEET_TRANSAKSI = "Transaksi"; const SHEET_PENGATURAN = "Pengaturan"; const FOLDER_BUKTI_ID = "11MlrKJh3vrP2s3xSiRj94ogLuv33qc"; // Folder ID Google Drive Anda /** * FUNGSI GET: Menangani semua permintaan pengambilan data */ function doGet(e) { var action = e.parameter.action; var ss = SpreadsheetApp.getActiveSpreadsheet(); // 1. Ambil Data Produk untuk Katalog Kasir if (action == 'getProducts') { var sheet = ss.getSheetByName(SHEET_PRODUK); var data = sheet.getDataRange().getValues(); var headers = data[0]; var result = []; for (var i = 1; i < data.length; i++) { var obj = {}; for (var j = 0; j < headers.length; j++) { var key = headers[j].toString().replace(/ /g, "_"); obj[key] = data[i][j]; } result.push(obj); } return translateToJson(result); } // 2. Cek Hasil Scan dari HP (Remote Scanner) if (action == 'checkScan') { var sheet = ss.getSheetByName(SHEET_PENGATURAN); var code = sheet.getRange("B10").getValue(); // Membaca cell B10 if (code && code !== "" && code !== "-") { sheet.getRange("B10").setValue("-"); // Langsung kosongkan setelah dibaca return translateToJson({ barcode: code }); } return translateToJson({ barcode: "-" }); } // 3. Ambil Stok Menipis (Untuk Dashboard Owner) if (action == 'getLowStock') { var sheet = ss.getSheetByName(SHEET_PRODUK); var data = sheet.getDataRange().getValues(); var lowStockItems = []; for (var i = 1; i < data.length; i++) { var stMin = data[i][5]; // Kolom F var stSkr = data[i][6]; // Kolom G if (stSkr <= stMin) { lowStockItems.push({ nama: data[i][1], stok: stSkr, min: stMin }); } } return translateToJson(lowStockItems); } // 4. Ambil Pengaturan Toko if (action == 'getSettings') { var sheet = ss.getSheetByName(SHEET_PENGATURAN); var settings = {}; if (sheet) { var data = sheet.getDataRange().getValues(); for (var i = 1; i < data.length; i++) { settings[data[i][0]] = data[i][1]; } } return translateToJson(settings); } } /** * FUNGSI POST: Menangani Simpan Transaksi & Kirim Scan dari HP */ function doPost(e) { try { var postData = JSON.parse(e.postData.contents); var ss = SpreadsheetApp.getActiveSpreadsheet(); // A. JIKA ACTION ADALAH KIRIM SCAN DARI HP if (postData.action == 'sendScan') { var sheetSetting = ss.getSheetByName(SHEET_PENGATURAN); sheetSetting.getRange("B10").setValue(postData.barcode); return translateToJson({ status: "success", message: "Barcode terkirim ke laptop" }); } // B. JIKA ACTION ADALAH SIMPAN TRANSAKSI var sheetSales = ss.getSheetByName(SHEET_TRANSAKSI); var sheetStock = ss.getSheetByName(SHEET_PRODUK); var timestamp = new Date(); var dateStr = Utilities.formatDate(timestamp, "GMT+8", "yyyy-MM-dd"); var timeStr = Utilities.formatDate(timestamp, "GMT+8", "HH:mm:ss"); var invoiceId = "TR-" + timestamp.getTime(); // 1. Upload Foto Bukti ke Drive (Jika Non-Tunai) var fileUrl = "-"; if (postData.imageContent && postData.method !== 'Cash') { var folder = DriveApp.getFolderById(FOLDER_BUKTI_ID); var blob = Utilities.newBlob(Utilities.base64Decode(postData.imageContent), "image/jpeg", "BUKTI_" + invoiceId + ".jpg"); var file = folder.createFile(blob); fileUrl = file.getUrl(); } // 2. Iterasi Keranjang: Simpan Baris & Potong Stok postData.cart.forEach(function(item) { // Simpan ke Transaksi (ID, Tgl, Jam, Barcode, Nama, Qty, Harga, Total, Kasir, Metode, Status, Link Bukti) sheetSales.appendRow([ invoiceId, dateStr, timeStr, item.Barcode, item.Nama_Barang, item.Qty, item.Harga_Jual, (item.Qty * item.Harga_Jual), postData.admin, postData.method, "Selesai", fileUrl ]); // Update Stok di Sheet Produk (Kolom G) var stockData = sheetStock.getDataRange().getValues(); for (var k = 1; k < stockData.length; k++) { if (stockData[k][0].toString() == item.Barcode.toString()) { var s_skr = stockData[k][6]; sheetStock.getRange(k + 1, 7).setValue(s_skr - item.Qty); break; } } }); return translateToJson({ status: "success", invoice: invoiceId }); } catch (err) { return translateToJson({ status: "error", message: err.toString() }); } } /** * HELPER: Mengubah data ke format JSON */ function translateToJson(data) { return ContentService.createTextOutput(JSON.stringify(data)).setMimeType(ContentService.MimeType.JSON); }
